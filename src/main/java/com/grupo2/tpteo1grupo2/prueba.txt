DECLARE.SECTION
[var1, var2, mensaje] := [INT, FLOAT, STRING]
[contador, limite] := [INT, INT]
ENDDECLARE.SECTION

PROGRAM.SECTION
//* Asignaciones numericas y de string *//
var1 ::= 15
var2 ::= 99.9
mensaje ::= "Hola Mundo"

//* Declaracion de un ciclo WHILE *//
WHILE (var1 < limite)
    contador ::= contador + 1
    IF (contador >= 10) THEN
        write "Contador alcanzo el limite"
        mensaje ::= "Limite alcanzado"
    ELSE
        write contador //* Muestra el valor de contador *//
    ENDIF
ENDWHILE

//* Condiciones y operaciones con constantes bin *//
var3 ::= 0b1101
IF (var3 < 16 && var2 >= 10.5) THEN
    write var3
ENDIF

//* Salidas de prueba *//
write mensaje
write 123.45
write 0b1011
write var2
ENDPROGRAM.SECTION
